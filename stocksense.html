const searchInp = document.querySelector('.srch-inp'); const searchBtn = document.querySelector('.srch-btn'); const searchDrop = document.querySelector('.srch-drop'); // 1. Function to search for symbols as you type searchInp.addEventListener('input', async (e) => { const query = e.target.value; if (query.length < 2) { searchDrop.classList.remove('open'); return; } const res = await fetch(`${window.location.origin}/api/market?action=search&q=${query}`); const data = await res.json(); if (data.results && data.results.length > 0) { searchDrop.innerHTML = data.results.map(s => `
${s.symbol} ${s.name}
`).join(''); searchDrop.classList.add('open'); } }); // 2. Function to fetch and display the selected stock async function selectStock(symbol) { searchDrop.classList.remove('open'); searchInp.value = symbol; const res = await fetch(`${window.location.origin}/api/market?action=quote&symbol=${symbol}`); const stock = await res.json(); if (stock.price) { // This updates the "Dark Horse" or Main Card with the new data const dhSym = document.querySelector('.dh-sym'); const dhName = document.querySelector('.dh-name'); const dhPrice = document.querySelector('.dh-mv'); // Target the price spot if(dhSym) dhSym.innerText = stock.symbol; if(dhName) dhName.innerText = stock.longName; if(dhPrice) dhPrice.innerText = "₹" + stock.price.toLocaleString('en-IN'); alert(`Loaded ${stock.longName}: ₹${stock.price}`); } }
