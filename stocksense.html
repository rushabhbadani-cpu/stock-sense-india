// --- STOCKSENSE CORE LOGIC V2 --- const API_URL = '/api/market'; let currentStock = null; // 1. Setup Search (Home & Research) function setupSearch(inpId, dropId, onSelect) { const inp = document.getElementById(inpId); const drop = document.getElementById(dropId); let debounce; inp.addEventListener('input', () => { clearTimeout(debounce); const q = inp.value.trim(); if (q.length < 2) return drop.classList.remove('open'); debounce = setTimeout(async () => { try { const res = await fetch(`${API_URL}?action=search&q=${q}`); const data = await res.json(); drop.innerHTML = data.results.map(s => `
${s.symbol} ${s.name}
`).join(''); drop.classList.add('open'); } catch (e) { console.error("Search failed"); } }, 300); }); } async function handleSelect(sym, inpId, dropId) { document.getElementById(dropId).classList.remove('open'); document.getElementById(inpId).value = sym; showPage('research'); loadStock(sym); } // 2. Load Stock & Calculate Educational Score async function loadStock(sym) { const rCard = document.querySelector('.rcard'); rCard.classList.add('v'); rCard.innerHTML = '
Fetching live market data...
'; try { const res = await fetch(`${API_URL}?action=quote&symbol=${sym}`); const s = await res.json(); currentStock = s; // Calculate Scorecard (Accountant Style Logic) let score = 0; if (s.pe > 0 && s.pe < 25) score += 40; // Value if (s.changePct > 0) score += 30; // Momentum if (s.price > (s.high52 * 0.8)) score += 30; // Strength rCard.innerHTML = `
${s.symbol}
${s.name}
${s.sector}
${score}
SCORE
PRICE
₹${s.price.toFixed(2)}
CHG%
${s.changePct.toFixed(2)}%
52W H
₹${s.high52.toFixed(0)}
P/E
${s.pe.toFixed(1)}

Education Insight: This stock has a score of ${score}/100 based on fundamental weightage. A higher score indicates better alignment with historical growth patterns.

`; } catch (e) { rCard.innerHTML = '
Error loading stock. Please try a different symbol.
'; } } // 3. Navigation function showPage(p) { document.querySelectorAll('.page').forEach(pg => pg.classList.remove('act')); document.getElementById(p + 'Page').classList.add('act'); document.querySelectorAll('.nitem').forEach(ni => ni.classList.remove('on')); document.querySelector(`[onclick="showPage('${p}')"]`).classList.add('on'); } // Initialize document.addEventListener('DOMContentLoaded', () => { setupSearch('homeSearch', 'homeDrop'); setupSearch('resSearch', 'resDrop'); showPage('home'); });
