async function loadStock(sym) { if(!sym) return; showPage('research'); // Switch to your original research tab // UI Loading State (Your CSS classes) const priceEl = document.querySelector('.res-price'); if(priceEl) priceEl.innerText = "Syncing..."; try { // 1. Get Live Price & Analysis Data const pRes = await fetch(`/api/market?action=quote&symbol=${sym}`); const p = await pRes.json(); // Populate your original UI fields document.getElementById('resSym').innerText = sym; document.getElementById('resName').innerText = p.name || sym; document.getElementById('resPrice').innerText = `₹${p.price || 'N/A'}`; // Update Portfolio Logic (If item is in portfolio) updatePortfolioPrice(sym, p.price); // Populate Analysis Tab (Key Statistics) const analysisHtml = `
P/E Ratio
${p.pe}
Div. Yield
${p.yield}
Market Cap
${p.marketCap}
Revenue
${p.revenue}
`; const analysisTab = document.getElementById('analysisTab'); // Match your tab ID if(analysisTab) analysisTab.innerHTML = analysisHtml; // 2. Get Corporate Ledger (The Board Meetings you need) const cRes = await fetch(`/api/market?action=corporate&symbol=${sym}`); const c = await cRes.json(); let corpTable = ''; const meetings = c.board_meetings?.data || []; meetings.slice(0, 10).forEach(m => { corpTable += ``; }); const corporateTab = document.getElementById('corporateTab'); // Match your tab ID if(corporateTab) corporateTab.innerHTML = corpTable + '
${m[0]} ${m[1]}
'; } catch (e) { console.error("Audit Failed", e); } }
